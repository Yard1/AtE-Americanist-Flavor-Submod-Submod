		set_title_flag = e_usa_senator_title
		if = {
			limit = { realm = { title = e_usa } }
			de_jure_liege = e_usa
		}
		holder_scope = {
			if = {
				limit = { 
					liege = { character = ROOT }
					OR = {
						has_minor_title = title_councilmember_emperor
						has_minor_title = title_councilmember_king
					}
				}
				liege = {
					set_character_flag = swapping_councillors
				}
				remove_title = title_councilmember_emperor
				remove_title = title_councilmember_king
			}
			if = {
				limit = { is_voter = no }
				give_minor_title = title_senator_k_new_york
			}
			liege = {
				clr_character_flag = swapping_councillors
			}
